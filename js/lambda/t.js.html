/*
 *
INCREMENT = -> n { -> p { -> x { p[n[p][x]] } } }
DECREMENT = -> n { -> f { -> x { n[-> g { -> h { h[g[f]] } }]
                                  [-> y { x }][-> y { y }] } } }
*/

/*
ADD      = -> m { -> n { n[INCREMENT][m] } }
SUBTRACT = -> m { -> n { n[DECREMENT][m] } }
MULTIPLY = -> m { -> n { n[ADD[m]][ZERO] } }
POWER    = -> m { -> n { n[MULTIPLY[m]][ONE] } }
*/

INCREMENT = function(n) {
    return function(f) {
        return function(x) {
            f(n(f)(x))
        }
    }
}

ADD = function(m) {
    return function(n) {
        n(INCREMENT)(m)
    }
}
